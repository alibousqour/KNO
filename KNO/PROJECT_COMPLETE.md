### The Setup
```python
from kno_utils import edex_monitor

class Agent:
    def __init__(self):
        self.monitor = edex_monitor
    
    async def process(self, input):
        await self.monitor.update_status(
            agent_status="THINKING",
            current_task="Processing...",
            progress=50
        )
        # ... do work ...
        await self.monitor.update_status(
            agent_status="IDLE",
            progress=100
        )
```
